\olsection{Three valued logic: tableaux}

\begin{prob}[Tableaus]
Use tableaux to say which of the following hold. If incorrect, give a counterexample.
\begin{enumerate}
    \item $\Obj{\lnot\lnot p \Proves[\LogKs] p}$
	\item $\Obj{\Proves[\LogKs] p \lor\lnot p}$
	\item $\Obj{\Proves[\LogKw] \lnot(p \land\lnot p)}$
    \item $\Obj{p, p \lif q \Proves[\LogLuk[3]] q}$
    \item $\Obj{p, p \lif q \Proves[\LogLP] q}$
    \item $\Obj{\Proves[\LogKs] p \rightarrow p}$
    \item $\Obj{\Proves[\LogLP] p \rightarrow p}$
    \item $\Obj{p, \lnot p \Proves[\LogKs] q}$
    \item $\Obj{p, \lnot p \Proves[\LogLP] q}$
\end{enumerate}
\end{prob}
